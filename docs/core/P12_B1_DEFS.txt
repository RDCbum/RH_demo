def ERURH.ERU_mode_beta : ℝ → Prop :=
fun β => ∃ s0, ∀ {s : ℝ}, s ≥ s0 → |ERURH.logR_alpha s| ≥ Real.exp ((β - 1 / 2) * s)
def ERURH.logR_alpha : ℝ → ℝ :=
fun s =>
  have x := Real.exp s;
  ERURH.AlphaPsi.E x / x
