theorem ERURH.RH_from_ERURH_conditional : ∀ (G : ERURH.ERURH_GlobalAssumptions) (hAxioms : ERURH.AxiomsShimAccepted),
  ERURH.CertificatesCorrectAlpha → ERURH.NumericCoverageAlpha G.window.ctx → ERURH.RiemannHypothesis ERURH.xiAlpha :=
fun G hAxioms hCerts hNumeric =>
  ERURH.ERURH_GlobalAssumptions.casesOn (motive := fun x =>
    ERURH.NumericCoverageAlpha x.window.ctx → ERURH.RiemannHypothesis ERURH.xiAlpha) G
    (fun eru window hNumeric =>
      ERURH.WindowScalingAssumptions.casesOn (motive := fun x =>
        ERURH.NumericCoverageAlpha { eru := eru, window := x }.window.ctx → ERURH.RiemannHypothesis ERURH.xiAlpha)
        window
        (fun ctx hA1 hLow hTail hNumeric =>
          ERURH.RH_from_ERURH_assumptions eru ctx hA1 hLow hTail hAxioms hCerts hNumeric)
        hNumeric)
    hNumeric
